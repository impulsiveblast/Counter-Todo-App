<script>

let entries =
JSON.parse(
localStorage.getItem("applicationTracker")
) || [];

const list =
document.getElementById("list");

const form =
document.getElementById("form");

document.getElementById("todayDate").innerText =
new Date().toDateString();

document.getElementById("addNew")
.addEventListener("click",()=>{

  form.style.display =
  form.style.display === "block"
  ? "none"
  : "block";

});

document.getElementById("saveBtn")
.addEventListener("click",()=>{

  let title =
  document.getElementById("titleInput")
  .value.trim();

  let count =
  document.getElementById("countInput")
  .value;

  if(title === ""){
    title = "Total Applications";
  }

  if(count === ""){
    count = 0;
  }

  entries.push({
    id:Date.now(),
    title:title,
    count:Number(count),
    date:new Date().toLocaleDateString()
  });

  save();
  render();

  document.getElementById("titleInput").value="";
  document.getElementById("countInput").value="";

  form.style.display="none";
});

function save(){
  localStorage.setItem(
    "applicationTracker",
    JSON.stringify(entries)
  );
}

function render(){

  list.innerHTML="";

  let total = 0;

  if(entries.length === 0){
    list.innerHTML =
    '<div class="empty">No trackers added yet.</div>';
  }

  entries.forEach(item=>{

    total += item.count;

    const card =
    document.createElement("div");

    card.className="card";

    card.innerHTML = `
      <h3>${item.title}</h3>

      <div class="count">
        ${item.count}
      </div>

      <div class="actions">

        <button class="plus">➕</button>

        <button class="minus">➖</button>

        <button class="reset">🔄</button>

        <button class="delete">🗑️</button>

      </div>

      <div class="date">
        Saved on: ${item.date}
      </div>
    `;

    const buttons =
    card.querySelectorAll("button");

    buttons[0].onclick = ()=>{

      item.count++;

      save();
      render();
    };

    buttons[1].onclick = ()=>{

      if(item.count > 0){
        item.count--;
      }

      save();
      render();
    };

    buttons[2].onclick = ()=>{

      item.count = 0;

      save();
      render();
    };

    buttons[3].onclick = ()=>{

      entries =
      entries.filter(
        e => e.id !== item.id
      );

      save();
      render();
    };

    list.appendChild(card);

  });

  document.getElementById("totalCount")
  .innerText = total;
}

render();

</script>
